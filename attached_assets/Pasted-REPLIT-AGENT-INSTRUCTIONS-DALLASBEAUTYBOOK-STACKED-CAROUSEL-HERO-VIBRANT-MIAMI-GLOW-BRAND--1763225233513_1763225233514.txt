import React, { useState } from "react";

type SectionCard = {
  id: string;
  title: string;
  subtitle: string;
  ctaLabel: string;
  href: string;
  gradient: string;
};

const SECTIONS: SectionCard[] = [
  {
    id: "discover",
    title: "Discover DFW Beauty",
    subtitle:
      "Explore top med spas, lash artists, salons, and skincare studios across Dallas–Fort Worth.",
    ctaLabel: "Explore Businesses",
    href: "/explore",
    gradient: "linear-gradient(135deg,#FD3A84,#FF7E5F)",
  },
  {
    id: "community",
    title: "Join the Community",
    subtitle:
      "Ask questions, share tips, and get real recommendations from locals and professionals.",
    ctaLabel: "Visit Community",
    href: "/community",
    gradient: "linear-gradient(135deg,#2EDFD0,#B3FFF7)",
  },
  {
    id: "owners",
    title: "For Business Owners",
    subtitle:
      "Claim your listing, post updates, and grow your bookings with zero upfront cost.",
    ctaLabel: "Claim Your Listing",
    href: "/claim",
    gradient:
      "linear-gradient(135deg,#FF7E5F,#FD3A84,#102646)",
  },
  {
    id: "suppliers",
    title: "Supplier Marketplace",
    subtitle:
      "Connect with brands and vendors for products, equipment, and wholesale deals.",
    ctaLabel: "Browse Suppliers",
    href: "/vendors",
    gradient: "linear-gradient(135deg,#102646,#364A8A,#2EDFD0)",
  },
];

export const SectionCarousel: React.FC = () => {
  const [activeId, setActiveId] = useState(SECTIONS[0].id);

  return (
    <section className="relative min-h-[80vh] bg-[#FFF1ED] flex items-center justify-center overflow-hidden px-4">
      <div className="relative flex w-full max-w-6xl flex-col lg:flex-row gap-8 items-center z-10">
        <div className="relative flex-1 h-[420px] flex items-center justify-center">
          {SECTIONS.map((section, index) => {
            const isActive = section.id === activeId;
            const activeIndex = SECTIONS.findIndex(s => s.id === activeId);
            const offset = index - activeIndex;

            return (
              <button
                key={section.id}
                onClick={() => setActiveId(section.id)}
                className="absolute inset-0 max-w-xl rounded-[30px] p-8 text-left text-white shadow-2xl transition-all duration-300"
                style={{
                  backgroundImage: section.gradient,
                  transform: `
                    translateX(${offset * 24}px)
                    translateY(${offset * 18}px)
                    scale(${isActive ? 1 : 0.92 - Math.abs(offset) * 0.03})
                    rotate(${offset * 2}deg)
                  `,
                  opacity: isActive ? 1 : 0.8 - Math.abs(offset) * 0.1,
                  zIndex: 20 - Math.abs(offset),
                }}
              >
                <h2 className="text-3xl font-extrabold md:text-4xl">{section.title}</h2>
                <p className="mt-3 text-white/90">{section.subtitle}</p>

                <a
                  href={section.href}
                  className="inline-flex mt-6 rounded-full bg-white/90 text-[#102646] px-5 py-2.5 font-semibold shadow-lg"
                >
                  {section.ctaLabel} ↗
                </a>
              </button>
            );
          })}
        </div>

        <div className="flex-1 max-w-md">
          <h1 className="text-4xl font-extrabold text-[#102646]">
            One platform. Every corner of DFW beauty.
          </h1>
          <p className="mt-4 text-[#434960]">
            Tap through each card to preview every part of the platform—from discovery
            and reviews to the marketplace and community Q&A.
          </p>
        </div>
      </div>
    </section>
  );
};
